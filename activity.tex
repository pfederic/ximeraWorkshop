 \begin{problem}
   We can add such as in $2 + 2 = \answer{4}$.
   \begin{problem}
     Multiplication looks like $3 \times 3 = \answer{9}$.
     \begin{problem}
       Now consider $\sqrt{\answer{16}} = 4$.
       \begin{problem}
         Therefore $4 \times 4 = \answer{16}$.
       \end{problem}
     \end{problem}
   \end{problem}
   \end{problem}
